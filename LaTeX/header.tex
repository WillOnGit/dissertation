% Proof environment
\newenvironment{proof}
 {
   \begin{trivlist}
     \item[] {\bf Proof.}
 }
 {
    \nolinebreak
    \hfill
    \rule{2mm}{2mm}
   \end{trivlist}
 }
 
% Sketch of proof environment
\newenvironment{sproof}
{
   \begin{trivlist}
     \item[] {\bf Sketch of proof.}
 }
{
    \nolinebreak
    \hfill
    \rule{2mm}{2mm}
   \end{trivlist}
}
